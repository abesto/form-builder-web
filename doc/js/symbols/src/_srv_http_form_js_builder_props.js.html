  9 /**
 10  * @class Egy tulajdonságot leíró objektum
 11  *
 12  * @param name A tulajdonság neve; fordításnál használjuk
 13  * @param dom A tulajdonság beállításaira szolgáló DOM
 14  */
 15 function Prop(name, dom)
 16 {
 17   this.name = name;
 18   this.dom = dom;
 19   /** Az utolsó megjelenített Prop, aminél .focus == true megkapja a fókuszt */
 20   this.focus = false;
 21 
 22   /** @return A megjelenítendő elem JQuery wrapperben */
 23   this.render = function() { return $(this.dom); };
 24 }
 25 
 26 /**
 27  * @class Prop objektumok gyűjteménye
 28  */
 29 function PropsCollection()
 30 {
 31   /** A tulajdonság-csoportok neveinek listája */
 32   this.groups = Array();
 33   /** A tulajdonság-csoportok [csoportnév: [{@link Prop}1, {@link Prop}2, ...], ...] asszociatív tömbje */
 34   this.props = new Object();
 35 
 36   /**
 37    * Új tulajdonság-csoport létrehozása
 38    *
 39    * @param name A tulajdonság-csoport neve
 40    */
 41   this.add_group = function(name) {
 42     this.groups.push(name);
 43     this.props[name] = Array();
 44   };
 45 
 46   /**
 47    * Tulajdonság hozzáadása egy csoporthoz
 48    *
 49    * @param group A tulajdonság-csoport, amihez hozzáadjuk a tulajdonságot
 50    * @param prop Prop objektum
 51    * @param focus Nem kötelező, ld. {@link Prop#focus}
 52    */
 53   this.add_prop = function(group, prop, focus)
 54   {
 55     if (focus) prop.focus = true;
 56     this.props[group].push(prop);
 57   };
 58 
 59   /** @return A tulajdonság-csoportok listája */
 60   this.get_groups = function() { return this.groups; };
 61   /**
 62    * @param group Egy tulajdonság-csoport neve
 63    * @return A csoporthoz tartozó {@link Prop} tulajdonságok listája
 64    */
 65   this.get_props = function(group) { return this.props[group]; };
 66 
 67   /**
 68    * Az adott tulajdonság-csoporthoz tartozó tualjdonságok listájának felülírása
 69    * @param name A tulajdonság-csoport
 70    * @param props Az új {@link Prop}-lista
 71    */
 72   this.set_group = function(name, props) { this.props[name] = props; };
 73 }

 75 /**
 76  * @namespace Az elemek beállítható tulajdonságaihoz tartozó input elemeket létrehozó függvények<br />
 77  */
 78 const PROPS = {
 79   /**
 80    * td elem típusa
 81    * @return Radio inputokat tartalmazó span; lehetséges értékei: text (szöveg) és input (beviteli mező)
 82    */
 83   td_type: function()
 84   {
 85     var span = $('<span>');
 86     var selected = check_selected_type('td');
 87     var types = Array('text', 'input|select');
 88     var child = selected.firstChild;
 89     var child_type = '';
 90     if (child) child_type = child.nodeName.toLowerCase().replace('#', '');
 91     else child_type = 'text';
 92     for (var i = 0; i < types.length; i++) {
 93       var type = types[i];
 94       var radio = $('<input type="radio" name="td_type" />').attr({'value': types[i], 'onchange': 'set_td_type(this.value)', 'id': type});
 95       var label = document.createTextNode(trans.td_types[types[i]]);
 96       if (types[i].match(child_type)) radio.attr('checked', 'true');
 97       span.append(radio).append(
 98         $('<label>').attr('for', type).append(label)
 99       );
100     }
101     return span;
102   },
103 
104   /**
105    * Az input elem típusa
106    *
107    * @return Select elem
108    */
109   input_type: function()
110   {
111     var select = document.createElement('select');
112     var selected = check_selected_type('td');
113     select.setAttribute('name', 'input_type');
114     var types = Array('text', 'button', 'checkbox', 'radio', 'file', 'select');
115     var type = selected.firstChild.getAttribute('type').toLowerCase();
116     for (var i = 0; i < types.length; i++) {
117       var option = document.createElement('option');
118       option.setAttribute('value', types[i]);
119       option.appendChild( document.createTextNode(trans.td_types[types[i]]) );
120       if (type == types[i]) option.setAttribute('selected', 'true');
121       select.appendChild(option);
122     }
123     select.setAttribute('onchange', 'set_input_type(this.value)');
124     return select;
125   },
126 
127   /**
128    * Beviteli elem name tulajdonságához kapcsolt mező
129    *
130    * @return Input elem
131    */
132   name: function() {
133     var input = make_input('name', 'text');
134     input.setAttribute('onchange', "set_name($('.selected')[0], this.value)");
135     input.value = $('.selected')[0].firstChild.getAttribute('name');
136     return input;
137   },
138 
139   /**
140    * Fieldset elem legend gyermekének szövegéhez kapcsolt mező
141    *
142    * @return Input elem
143    */
144   legend: function() {
145     var input = make_input('legend', 'text');
146     input.setAttribute('onchange', "$('.selected')[0].firstChild.innerHTML = this.value");
147     input.value = $('.selected')[0].firstChild.innerHTML;
148     return input;
149   },
150 
151   /**
152    * A form nevének beállítása; szerkesztéskor H1-ként jelenik meg.
153    *
154    * @return Input elem
155    */
156   form_name: function() {
157     var input = make_input('text', 'form_name');
158     input.setAttribute('onchange', "$('#title').html(this.value)");
159     input.value = $('#title').html();
160     return input;
161   },
162 
163   /**
164    * A select elem egy option-jéhez tartozó text mező és eltávolító gomb
165    *
166    * @param num Az option 0-alapú sorszáma
167    * @param text A kezdeti szövegtartalom
168    * @return Div elem
169    */
170   option: function(num, text) {
171     var $option = $('<div>').append(
172       make_input('option_'+num, 'text')
173     ).append(
174       make_input('remove_'+num, 'button')
175     );
176 
177     $option.children('input[type=text]').attr('value', text).change(
178       function () {
179         with ($('.selected select')) {
180           if (children('option').length <= num)
181             append($('<option>'));
182           children('option')[num].innerHTML = this.value;
183         }
184         update_props('td');
185       });
186 
187     $option.children('input[type=button]').attr('value', '-').click(
188       function() {
189         var $elem = $($('.selected option')[num]);
190         if ($('.selected option').length == 1)
191           $elem.html('');
192         else
193           $elem.remove();
194         update_props('td');
195       });
196     return $option[0];
197   }
198 };

200 /**
201  * @namespace Az egyes elemekhez tartozó tulajdonságokat visszaadó függvények
202  * az elem tartalmát figyelembe véve
203  *
204  * @return PropsCollection
205  */
206 const props = {
207   /** @return PropsCollection */
208   td: function ()
209   {
210     var selected = check_selected_type('td');
211     var ret = new PropsCollection();
212     ret.add_group('td');
213     ret.add_prop('td', new Prop('td_type', PROPS.td_type()));
214 
215     var type = get_td_type(selected);
216     var input_type = get_input_type(selected);
217     if (type == 'input') {
218       ret.add_group('input');
219       ret.add_prop('input', new Prop('name', PROPS.name()));
220       ret.add_prop('input', new Prop('input_type', PROPS.input_type()));
221     }
222 
223     var text = make_input('td_text', 'text');
224     text.value = get_td_text(selected);
225     $(text).change(function () {set_td_text($('.selected')[0], this.value);});
226     if (input_type == 'file') {
227       text.disabled = true;
228       text.value = trans.no_file_label;
229     }
230 
231     if (type != 'select')
232       ret.add_prop('td', new Prop('td_text', text), true);
233     else {
234       ret.add_group('options');
235       with (selected.firstChild) {
236         for (var i = 0; i < options.length; i++)
237           ret.add_prop('options', new Prop('option', PROPS.option(i, options[i].innerHTML)), true);
238         if (options[0].innerHTML != '')
239           ret.add_prop('options', new Prop('option', PROPS.option(options.length, '')), true);
240       }
241     }
242 
243     return ret;
244   },
245 
246   /**
247    * Fieldsetnél be lehet állítani a legend feliratát
248    * @return PropsCollection
249    */
250   fieldset: function()
251   {
252     var selected = check_selected_type('fieldset');
253     var text = selected.firstChild.innerHTML;
254     var ret = new PropsCollection();
255     ret.add_group('fieldset');
256     ret.add_prop('fieldset', new Prop('legend', PROPS.legend()), true);
257     return ret;
258   },
259 
260   /**
261    * Formnak lehet neve; H1-ként jelenik meg szerkesztés közben
262    * @return PropsCollection
263    */
264   form: function()
265   {
266     var selected = check_selected_type('form');
267     var ret = new PropsCollection();
268     ret.add_group('form');
269     ret.add_prop('form', new Prop('form_name', PROPS.form_name()), true);
270     return ret;
271   }
272 };
