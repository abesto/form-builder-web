 13 /**
 14  * A műveleteket tartalmazó form frissítése
 15  */
 16 function update_actions(type)
 17 {
 18   // Kitöröljük a régi tartalmat
 19   var bar = $('#actions_fs');
 20   $('#actions_fs input, #actions_fs span, #actions_fs br').remove();
 21   if (type == 'none') return;
 22 
 23   var current = actions[type];
 24   if (!current) {
 25     // Ha nincs végezhető művelet, ezt tudatjuk a felhasználóval, és végeztünk
 26     bar.append(node_with_text('span', trans.no_actions(type)));
 27     return;
 28   }
 29   current = current();
 30 
 31   // Létrehozzuk a műveletekhez tartozó gombokat
 32   for (var i = 0; i < current.length; i++) {
 33     var name = current[i];
 34     if (name == 'br') {
 35       bar.append($('<br />'));
 36       continue;
 37     }
 38     bar.append(
 39       $('<input>').attr({'type': 'button', 'name': name, 'value': trans.actions[name], 'onclick': 'ACTIONS.'+name})
 40     );
 41   }
 42 }

 44 /**
 45  * A tulajdonságokat tartalmazó form frissítése
 46  */
 47 function update_props(type)
 48 {
 49   // Kitöröljük a régi tartalmat
 50   $('#props_form fieldset').remove();
 51   $('#props span').remove();
 52   var form = $('#props_form');
 53   if (type == 'none') return;
 54 
 55   var current = props[type];
 56   if (!current) {
 57     // Ha nincs beállítható tulajdonság, ezt tudatjuk a felhasználóval, és végeztünk
 58     $('#props_form').append(node_with_text('span', trans.no_props(type)));
 59     return;
 60   }
 61   current = current();
 62 
 63   // Létrehozzuk a tulajdonságokhoz tartozó beviteli mezőket
 64   var groups = current.get_groups();
 65   for (var groupi = 0; groupi < groups.length; groupi++) {
 66     var group = groups[groupi];
 67     var fs = $('<fieldset>').attr('id', group).append(
 68       $('<legend>').append(trans.prop_groups[group]));
 69     form.append(fs);
 70 
 71     var table = $('<table>');
 72     fs.append(table);
 73 
 74     var _props = current.get_props(group);
 75     for (var propsi = 0; propsi < _props.length; propsi++) {
 76       var prop = _props[propsi];
 77       var text = trans.prop_names[prop.name];
 78       if (text == undefined) text = prop.name;
 79 
 80       var $elem = prop.render();
 81       table.append(
 82         $('<tr>').append(
 83           $('<td>').append(text)
 84         ).append(
 85           $('<td>').append($elem)
 86         ));
 87       if (prop.focus)
 88         if ($elem[0].nodeName != "INPUT")
 89           $elem.children('input[type=text]').focus();
 90         else
 91           $elem.focus();
 92     }
 93   }
 94 }

 96 /**
 97  * Az eseményt indító elemet jelöli meg hoveredként
 98  */
 99 function hover()
100 {
101   $hovered.unbind('click');
102   $hovered.removeClass('hovered');
103   $(this).addClass('hovered').click(select_hovered);
104   $hovered = $(this);
105 }
106 
107 /**
108  * Az eseményt indító elem hovered-ségét megszűnteti
109  * Ha van kijelölhető szülője az elemnek, akkor azt jelöli meg hoveredként
110  */
111 function unhover()
112 {
113   $hovered.unbind('click');
114   $hovered.removeClass('hovered');
115   var name = $hovered[0].parentNode.nodeName;
116   if ((name != 'INPUT') && (name != 'TD') && (name != 'FORM') && (name != 'TABLE') && (name != 'FIELDSET'))
117   {
118     $hovered = $($hovered[0].parentNode);
119     while ((name == 'TBODY') || (name == 'TR')) {
120       $hovered = $($hovered[0].parentNode);
121       name = $hovered[0].nodeName;
122     }
123     $hovered.addClass('hovered').click(select_hovered);
124   }
125 }
126 
127 /**
128  * A hover és unhover eseménykezelők csak az éppen aktuálisan hovered
129  * elem click eseménykezelőjeként állítják be ezt, tehát 'this' a kattintást kapott elem
130  *
131  * Az ezzel megoldott probléma: ha table-nek és td-nek is van regisztrált click eseménykezelője,
132  * akkor table kezelője indul el, nem td-é.
133  */
134 function select_hovered()
135 {
136   $(this).select();
137 }
138 
139 /**
140  * Kijelöli a kapott elemet
141  */
142 $.fn.select = function ()
143 {
144   $('.selected').removeClass('selected');
145 
146   var $elem = $(this);
147   $elem.addClass('selected');
148   var type = $elem[0].nodeName.toLowerCase();
149   update_actions(type);
150   update_props(type);
151 };

153 /**
154  * Az UI nyelvének megváltoztatása
155  * @param id A nyelv indexe a TRANS tömbben
156  */
157 function set_lang(id)
158 {
159   id = parseInt(id.replace(/[^0-9]*/, ''));
160   trans = TRANS[id];
161   $('#actions_fs legend').html(trans.actions_label);
162 
163   var prop_groups = $('#props_form fieldset');
164   for (var i = 0; i < prop_groups.length; i++)
165     prop_groups[i].firstChild.innerHTML = trans.prop_groups[prop_groups[i].getAttribute('id')];
166 
167   var selected = $('.selected')[0];
168   if (selected) {
169     var name = selected.nodeName.toLowerCase();
170     update_actions(name);
171     update_props(name);
172   }
173 }

175 function make_html()
176 {
177   var text = htmlize($('#main form')[0], 0);
178   // tbody nem kell
179   text = text.replace(/ *<tbody>.*\n/g, '');
180   text = text.replace(/ *<\/tbody>.*\n/g, '');
181   // amit muszáj
182   text = text.replace(/</g, '<');
183   text = text.replace(/>/g, '>');
184   // megmutatjuk
185   var $html = $('#html');
186   $html.html('<pre>' + text + '</pre>').dialog('option', 'title', $('#title').html()+' HTML');
187   $html.dialog('open');;
188   if ($html.width() < 300) $html.css({'width': '300px'});
189 }
