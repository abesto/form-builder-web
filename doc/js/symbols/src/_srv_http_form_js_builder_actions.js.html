  9 // Tábla sor/cella műveletekhez irányok
 10 const LEFT  = 100;
 11 const RIGHT = 101;
 12 const UP    = 102;
 13 const DOWN  = 103;
 14 
 15 const WrongDir = new Error('Wrong direction');
 16 
 17 /**
 18  * @namespace A végezhető műveleteket tartalmazó objektum
 19  *
 20  * A tárol-append-select trükkre azért van szükség, mert a select függvény
 21  * elvárja, hogy már a DOM része legyen az elem
 22  */
 23 const ACTIONS = {
 24   /** Új fieldset táblázattal */
 25   add_fieldset: function()
 26   {
 27     var selected = check_selected_type('form');
 28     var $fieldset = $('<fieldset>').append($('<legend>'));
 29     $(selected).append($fieldset);
 30     $fieldset.select();
 31   },
 32 
 33   /** Fieldset törlése */
 34   remove_fieldset: function()
 35   {
 36     $(check_selected_type('fieldset')).remove();
 37     update_actions('none');
 38     update_props('none');
 39   },
 40 
 41   /** Új tábla */
 42   add_table: function()
 43   {
 44     var selected = check_selected_type(Array('form', 'fieldset'));
 45     var $td = $('<td>');
 46     $(selected).append($('<table>').append($('<tr>').append($td)));
 47     $td.select();
 48   },
 49 
 50   /** Tábla törlése */
 51   remove_table: function() { $(check_selected_type('table')).remove(); },
 52 
 53   /**
 54    * Létrehoz egy új cellát a kiválasztott cella mellett
 55    * @param dir Az új cella helye a kiválasztott cellához képest
 56    * @exception WrongDir Ha az irány nem LEFT vagy RIGHT
 57    */
 58   create_cell: function(dir)
 59   {
 60     var selected = check_selected_type('td');
 61     var tr = selected.parentNode;
 62     var td = $('<td>')[0];
 63     if      (dir == RIGHT) tr.insertBefore(td, selected.nextSibling);
 64     else if (dir == LEFT)  tr.insertBefore(td, selected);
 65     else    throw WrongDir;
 66     $(td).select();
 67     update_actions('td');
 68     update_props('td');
 69   },
 70 
 71   /**
 72    * Létrehoz egy új sort a táblázatban
 73    *
 74    * @param dir Az új sor helye a kiválasztott cella sorához képest
 75    * @exception WrongDir Ha az irány nem UP vagy DOWN
 76    */
 77   create_row: function(dir)
 78   {
 79     var selected = check_selected_type('td');
 80     var table = selected.parentNode.parentNode;
 81     var old = selected.parentNode;
 82     var $td = $('<td>');
 83     var tr = $('<tr>').append($td)[0];
 84     if      (dir == DOWN) table.insertBefore(tr, old.nextSibling);
 85     else if (dir == UP)   table.insertBefore(tr, old);
 86     else    throw WrongDir;
 87     $td.select();
 88   },
 89 
 90   /**
 91    * Két vízszintesen egymás mellett elhelyezkedő, azonos magasságú cellát
 92    * egyesít. Ha mindkét cella tartalma szöveg, akkor a tartalmakat
 93    * összekapcsolja egy szóközzel. Különben a kiválasztott cella tartalma
 94    * felülírja a másik celláét
 95    *
 96    * @param dir A cella-egyesítés iránya
 97    * @exception WrongDir Ha az irány nem LEFT vagy RIGHT
 98    */
 99   merge_cells: function(dir) {
100     var selected = check_selected_type('td');
101     var tr = selected.parentNode;
102     var victim;
103 
104     if (dir == LEFT) {
105       var prev = selected.previousSibling;
106       if ((get_td_type(selected) == 'text') && (get_td_type(prev) == 'text'))
107         prev.innerHTML += ' ' + selected.innerHTML;
108       else
109         prev.innerHTML = selected.innerHTML;
110       victim = selected;
111       selected = prev;
112       $(prev).addClass('selected');
113 
114     } else if (dir == RIGHT) {
115       var next = selected.nextSibling;
116       if ((get_td_type(selected) == 'text') && (get_td_type(next) == 'text'))
117         selected.innerHTML += ' ' + next.innerHTML;
118       victim = next;
119     } else throw WrongDir;
120     tr.removeChild(victim);
121 
122     var size0 = get_int_attrib(selected, 'colspan');
123     var size1 = get_int_attrib(victim, 'colspan');
124     selected.setAttribute('colspan', size0 + size1);
125 
126     update_actions('td');
127     update_props('td');
128   },
129 
130   /**
131    * Egy több oszlopos cellát oszt fel
132    * A cella tartalma a kiválasztott cellában marad
133    * Az új cella a megadott irányban jön létre, tartalom nélkül
134    *
135    * @param dir A cella-osztás iránya
136    * @exception WrongDir Ha az irány nem LEFT vagy RIGHT
137    */
138   split_cell: function(dir)
139   {
140     var td = document.createElement('td');
141     var selected = check_selected_type('td');
142     var tr = selected.parentNode;
143 
144     selected.setAttribute('colspan', get_int_attrib(selected, 'colspan')-1);
145     if      (dir == LEFT)  tr.insertBefore(td, selected);
146     else if (dir == RIGHT) tr.insertBefore(td, selected.nextSibling);
147     else    throw WrongDir;
148 
149     update_actions('td');
150     update_props('td');
151   },
152 
153   /** Cella törlése */
154   remove_cell: function()
155   {
156     var selected = check_selected_type('td');
157     var tr = selected.parentNode;
158     var table = tr.parentNode.parentNode;
159     $(selected).remove();  // <-- öngyilkosság :(
160     if (tr.childNodes.length == 0) $(tr).remove();  // <-- családostól :'(
161     if ($(table).find('tr').length == 0) $(table).remove();  // <-- tiszta lemmingek :(
162     update_actions('none');
163     update_props('none');
164   }
165 };
166 
167 /**
168  * @namespace Az egyes elemeken végezhető műveletek listáját visszaadó függvények a
169  *            kiválasztott elem helyzetét és környezetét figyelembevéve
170  */
171 const actions = {
172   /**
173    * td elemen végrehajtható műveletek<br />
174    * Mindig: cella létrehozása (jobbra, balra), új sor létrehozása (fel, le)<br />
175    * Ha van azonos magasságú szomszédos cella: cellák egyesítése (balra és/vagy jobbra)<br />
176    * Ha a cella több oszlopos: cella felosztása (balra, jobbra)
177    */
178   td: function()
179   {
180     var $sel = $(check_selected_type('td'));
181     var ret = Array('create_cell(LEFT)', 'create_cell(RIGHT)', 'create_row(UP)', 'create_row(DOWN)', 'br');
182 
183     if ($sel.prev().attr('rowspan') == $sel.attr('rowspan'))
184       ret.push('merge_cells(LEFT)');
185     if ($sel.next().attr('rowspan') == $sel.attr('rowspan'))
186       ret.push('merge_cells(RIGHT)');
187 
188     if (ret[ret.length-1] != 'br')
189       ret.push('br');
190 
191     if ($sel.attr('colspan') > 1) {
192       ret = ret.concat(Array('split_cell(LEFT)', 'split_cell(RIGHT)'));
193       ret.push('br');
194     }
195 
196     ret.push('remove_cell()');
197     return ret;
198   },
199 
200   /** Táblázatot csak törölni lehet. Szegény :( */
201   table: function()
202   {
203     check_selected_type('table');
204     return Array('remove_table()');
205   },
206 
207   /** A formhoz lehet hozzáadni táblázatot vagy fieldsetet */
208   form: function()
209   {
210     check_selected_type('form');
211     return Array('add_table()', 'add_fieldset()');
212   },
213 
214   /** A fieldsethez lehet hozzáadni táblázatot, vagy lehet törölni a fieldsetet */
215   fieldset: function()
216   {
217     check_selected_type('fieldset');
218     return Array('add_table()', 'remove_fieldset()');
219   }
220 };
