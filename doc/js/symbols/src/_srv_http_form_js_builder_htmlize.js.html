 12 /** A parser */
 13 function htmlize(el, level)
 14 {
 15   var name = el.nodeName.toLowerCase();
 16   if (name == 'h1') return '';
 17   var ret = '';
 18 
 19   if (name == 'tbody')
 20     level -= 1;
 21 
 22   if (name == "#text")
 23     ret += el.textContent;
 24   else {
 25     for (var i = 0; i < level; i++)
 26       ret += '  ';
 27     ret += "<" + name;
 28     var $el = $(el);
 29     var attrs = {'input': Array('type', 'name', 'id', 'checked'),  // Amiket át kell másolni, ha van
 30                  'td': Array('colspan'),
 31                  'label': Array('for')};
 32     if (attrs[name] != undefined)
 33       for (i = 0; i < attrs[name].length; i++) {
 34         var attr = attrs[name][i];
 35         if (($el.attr(attr) != undefined) && ($el.attr(attr) != ''))
 36           ret += ' ' + attr  + '="' + $el.attr(attr) + '"';
 37       }
 38     if ((name == 'input') && ($el.attr('type') != 'radio') && ($el.attr('type') != 'checkbox') && ($el.attr('value') != ""))
 39       ret += ' value="' + $el.attr('value') + '"';
 40     if (name == "input") {
 41       return ret + " />\n";
 42     }
 43     ret += ">";
 44   }
 45 
 46   if ((el.childNodes != undefined) && (el.childNodes.length > 0) && (el.childNodes[0].nodeName != '#text'))
 47     ret += "\n";
 48 
 49   for (var i = 0; i < el.childNodes.length; i++)
 50     ret += htmlize(el.childNodes[i], level+1);
 51 
 52   if (name != "#text") {
 53     if ((el.childNodes != undefined) && (el.childNodes.length > 0) && (el.childNodes[0].nodeName != '#text'))
 54       for (i = 0; i < level; i++)
 55         ret += '  ';
 56     ret += "</" + name + ">\n";
 57   }
 58   return ret;
 59 }
