 10 /**
 11  * Ellenőrzi a kiválasztott elem típusát
 12  * @param type A kívánt típus nodename-ja (pl. TD), vagy namejei Array-ban
 13  * @exception Ha az elem nem type típusú
 14  * @return A kiválasztott elem
 15  */
 16 function check_selected_type(type)
 17 {
 18   var selected = $('.selected')[0];
 19 
 20   if (typeof(type) == "string")
 21     type = Array(type);
 22 
 23   for (var i = 0; i < type.length; i++)
 24     if ($.nodeName(selected, type[i]))
 25       return selected;
 26 
 27   throw new Error(trans.wrong_node(type));
 28   return selected;
 29 }

 31 /**
 32  * Létrehoz egy szöveget tartalmazó elemet
 33  * @param name Az elem neve
 34  * @param text Az elem szövege
 35  * @example node_with_text('td', 'szöveg')    // <td>szöveg</td>
 36  * @return HTML
 37  */
 38 function node_with_text(name, text) { return $('<' + name + '>').append(text); }

 40 /**
 41  * Létrehoz egy input elemet
 42  * @param name Az elem name tulajdonsága
 43  * @param type Az input típusa
 44  * @example make_input('teszt', 'checkbox')    // <input type="checkbox" name="teszt" />
 45  * @return Az elem
 46  */
 47 function make_input(name, type) { return $('<input />').attr({'type': type, 'name': name})[0]; }

 49 /**
 50  * @param td A td elem
 51  * @return A td elem tartalmának típusa. 'text', 'input' vagy 'select'
 52  */
 53 function get_td_type(td) {
 54   if (!td.firstChild) return 'text';
 55   if (td.firstChild.nodeName == '#text') return 'text';
 56   if (td.firstChild.nodeName == 'INPUT') return 'input';
 57   if (td.firstChild.nodeName == 'SELECT') return 'select';
 58   return false;
 59 }

 61 /**
 62  * Beállítja a kapott cellában levő input mező name és id tulajdonságát.
 63  * Az id tulajdonság végére egy szám kerül, hogy az azonosító egyedi legyen.
 64  * A szám 1-től indul és a törlés miatt keletkező "lyukakat" feltölti
 65  *
 66  * @param td A td elem
 67  * @param name A name tulajdonság értéke
 68  */
 69 function set_name(td, name) {
 70   var $input = $(td.firstChild);
 71 
 72   // Ha üres a kapott name, töröljük a tulajdonságokat
 73   if (name == '') {
 74     $input.removeAttr('name').removeAttr('id');
 75     return;
 76   }
 77 
 78   // Megkeressük és beállítjuk az első szabad azonosítót
 79   var num = 1;
 80   $('#main input[id^=' + name + ']').each(
 81     function() {
 82       if ($(this).attr('id') == name + '_' + num)
 83         num += 1;
 84     });
 85   var id = name + '_' + num;
 86   $input.attr({'name': name, 'id': id});
 87   $input.next().attr('for', id); // Radio és checkbox elemek feliratához
 88 }

 90 /**
 91  * @param td Egy td elem
 92  * @return A cella beviteli mezőjének típusa
 93  */
 94 function get_input_type(td) {
 95   var input = td.firstChild;
 96   if (get_td_type(td) == 'text') return 'none';
 97   else return input.getAttribute('type');
 98 }

100 /**
101  * Megadja egy elem adott tulajdonságát int típusú változóként
102  * @param el Egy elem
103  * @param attrib A tulajdonság neve
104  * @return Ha a tulajdonság értéke null akkor 1, különben az érték intként
105  */
106 function get_int_attrib(el, attrib)
107 {
108   var val = el.getAttribute(attrib);
109   if (!val) return 1;
110   return parseInt(val);
111 }

113 /**
114  * @return A mező szöveges tartalma
115  */
116 function get_td_text(td)
117 {
118   var text = "";
119   var selected = check_selected_type('td');
120   var type = get_td_type(td);
121   // Szöveges tartalom
122   if (type == 'text') {
123     text = selected.innerHTML;
124   // Input mező
125   } else if (type == 'input') {
126     var intype = td.firstChild.getAttribute('type');
127     if ((intype == 'text') || (intype == 'button')) text = td.firstChild.value;
128     else if ((intype == 'radio') || (intype == 'checkbox')) text = td.textContent;
129   // Select mező
130   } else if (type == 'select') {
131     if (td.firstChild.firstChild)
132       text = td.firstChild.firstChild.innerHTML;
133     else
134       return '';
135   }
136   return text.replace(/^\s+|\s+$/g, '');  // Szóközöket eltávolítjuk az elejéről és a végéről
137 }

139 /**
140  * Az adott cella szöveges tartalmát állítja ba
141  * @param td A cella
142  * @param text A szöveg
143  */
144 function set_td_text(td, _text)
145 {
146   var type = get_td_type(td);
147   _text = _text.replace(/^\s+|\s+$/g, '');
148   if (type == 'text') {
149     td.innerHTML = _text;
150   } else if (type == 'input') {
151     type = td.firstChild.getAttribute('type');
152     if (td.childNodes.length == 2) td.removeChild(td.lastChild);
153     if      ((type == 'text') || (type == 'button')) td.firstChild.value = _text;
154     else if ((type == 'radio') || (type == 'checkbox')) {
155       var label = node_with_text('label', _text);
156       label.attr('for', td.firstChild.getAttribute('id'));
157       $(td).append(label);
158     }
159   } else if (type == 'select') {
160     with ($(td).children('select')) {
161       children().remove();
162       append('<option>'+_text+'</option>');
163     }
164   }
165 }

167 /**
168  * A kiválasztott cella input elemének típusát állítja be
169  * @param type Az elem type tulajsonsága
170  */
171 function set_input_type(type)
172 {
173   var td = check_selected_type('td');
174   var text = get_td_text(td);
175   var old_type = get_input_type(td);
176 
177   var $input = null;
178 
179   if ((old_type == 'select') && (type != 'select'))
180     $input = $('<input type="'+type+'">');
181   else if ((old_type != 'select') && (type == 'select'))
182     $input = $('<select>');
183 
184   if ((old_type != 'select') && (type != 'select')) {
185     $input = $(td.firstChild);
186     $input[0].setAttribute('type', type);
187   } else {
188     $(td).children().remove();
189     $(td).append($input);
190   }
191 
192   if (type == 'text') $input.attr('onchange', "update_props('td')");
193   else $input.attr('onchange', '');
194 
195   set_td_text(td, text);
196   // Fájl input esetén nincs felirat, ezért frissíteni kell a tulajdonságokat
197   // Selectnél pedig borul a szokásos felépítés
198   if ((type == 'file') || (old_type == 'file') || (old_type == 'select') || (type == 'select'))
199     update_props('td');
200 }

202 /**
203  * A kiválasztott td elem tartalmának típusát állítja be
204  * @param type Az elem tartalmának típusa
205  */
206 function set_td_type(in_type)
207 {
208   var selected = check_selected_type('td');
209   var text = get_td_text(selected);
210   if      (in_type == 'text') selected.innerHTML = '';
211   else if (in_type == 'input|select') {
212     var input = make_input('', 'text');
213     input.setAttribute('onchange', 'update_props(\'input\')');
214     selected.innerHTML = "";
215     selected.appendChild(input);
216   }
217   set_td_text(selected, text);
218   update_props('td');
219 }
